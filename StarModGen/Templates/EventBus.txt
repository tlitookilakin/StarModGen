namespace StardModGen.Utils;

///<summary>Automatically hooks up registered events and event handlers.</summary>
internal static partial class EventBus 
{
	///<summary>Attaches all handlers to their events. Should be called in Entry.</summary>
	///<param name="Helper">Your mod's ModHelper</param>
	public static partial void Register(StardewModdingAPI.IModHelper Helper)
	{
		{%- for Group in Handlers -%}
		{% if Group -%}
		{% for hook in Group %}
		{{Group}} += {{hook.owner}}.{{hook.method}};
		{%- endfor %}
		{%- endif %}
		{%- endfor %}
	}
}
